# U-boot configuration
loadaddr=0x82000000

# Kernel configuration
kernel_file=/bin/boot.bin
entry_point=0x80200000

# Commands
home_message=echo [WARNING] Ceci est un projet de TIPE réalisé par Thibault PIANA et Alan GARDIN, élèves de MP* au lycée thiers de Marseille 2015-2016

loadkernel=fatload mmc 0 ${loadaddr} ${kernel_file}
loadfiles=echo [INFORMATION] Download kernel on RAM...;run loadkernel
uenvcmd=run home_message;run loadfiles; echo [INFORMATION] Loading Kernel...;go ${entry_point}

# Remarque : Sous beaglebone, la commande principale est uenvcmd et non bootcmd !
